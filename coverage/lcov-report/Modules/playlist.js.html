const storage = require('node-persist')  // use basic file-based data persistence
const playlistConnection = require('./database').playlistConnection
const playlists = require ('./playlist.Spec')
 
storage.initSync()

exports.validate = function validate(req, res, next) {
	const playlist = req.body
	if (!playlist) return res.send(400, {Message: 'Need to send some data'})
	if (!playlist.id || !playlist.artist || !playlist.track) return res.send(400, {Message: 'Missing components: id, artist or track'})
	next()
}
 
exports.list = function list(req, res) {
	// get all the books saved in the user's list
	playlistConnection(req, res, playlists => res.send({playlist: playlists.values()}))
}
 
exports.add = function add(req, res) {
 
	// connect to the favourites DB then save the given book
	playlistConnection(req, res, playlists => {
 
		// first check that the id is not used already
		const list = req.body
 
		if (playlists.getItemSync(list.id)) return res.send(400, {message: 'id already exists', id: list.id})
 
		// TODO: omit the fields you don't want to save from book, before saving
 
		playlists.setItem(list.id, list, err => {
			if (err) {
				console.log(err)
				return res.send(500, {message: 'Could not add to playlist', list: list})
			} else {
				return res.send(201, {message: 'Added to playlist', list: list})
			}
		})
	})
}
 
exports.get = function get (req, res, next) {
	// use the ID in the URL to look up a particular favourite
	const trackid = req.params.id
 
	// connect to the favourites DB then get the requested book
	playlistConnection(req, res, playlists => {
		playlists.getItem(trackid, (err, book) => {
			if (err) return res.send(500, {message: 'Couldnt get the playlist'})
 
			// book will be undefined if there was no match
			if (!book) return res.send(404, {message: 'id not found', id: trackid})
			
			// otherwise return the book object
			res.send({favourite: trackid})
		})
	})
	next()
}
