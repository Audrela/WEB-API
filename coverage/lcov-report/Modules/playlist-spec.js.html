'use strict'
/*istanbul ignore next*/
/* global expect */
 
var list = require("./Modules/playlist")
const schema = require("./schema")
const lists = require("./lists")
 
describe('Playlist', function () {
  beforeEach( done => {
		schema.List.remove({}, err => {
			if (err) expect(true).toBe(false)
			new schema.List({name: 'colours', list: ['red', 'orange', 'yellow']}).save( (err, list) => {
				if (err) expect(true).toBe(false)
				schema.List.count({}, (err, count) => {
					if (err) expect(true).toBe(false)
					expect(count).toBe(1)
					done()
				})
			})
		})
	/* this code is run before each of the tests */
	beforeEach(() => {
		list.add('id')
	})
 
	/* this code is fun after EACH test */
	afterEach(() => {
		list.clear()
	})
  it('Should return the playlist', () => {
    const items = list.getAll()
    expect(items[0].id).toBe('3TVXtAsR1Inumwj472S9r4')
  })
     
	it('should add new song to playlist', () => {
		list.add('id1')
		expect(list.count()).toBe(3)
	})
  
  xit('Should delete ', () => {
    list.deleteId('3TVXtAsR1Inumwj472S9r4')
		expect(list.count()).toBe(0)
		const items = list.getAll()
		expect(items[0].id).toBe(0)
    
  })
  
  xit('should update the current list', () => {
    const items = list.getAll()
    list.clear()
    list.add('id')
    list.add('artist')
    expect(items[0].toBe('3TVXtAsR1Inumwj472S9r4'))
    expect(items[1].toBe('Drake'))
  })
    
})
})
