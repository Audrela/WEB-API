'use strict'
 
const playlist = require('./Modules/playlist')
 
exports.validate = (req, res, next) => {
	const plist = req.body
	playlist.validate(plist, (err, isValid) => {
		if (isValid) return next()
		return res.send(400, err)
	})
}
 
exports.list = (req, res) => {
	const username = req.authorization.basic.username
	playlist.list(username, (err, result) => {
		if (err) return res.send(400, err)
		return res.send({playlist: result})
	})
}
 
 
exports.add = (req, res) => {
	const username = req.authorization.basic.username
	const track = req.body
	playlist.add(username, track, (err, result) => {
		if (err) return res.send(400, err)
		return res.send(result)
	})
}
 
 
exports.get = (req, res) => {
	const username = req.authorization.basic.username
	const trackid = req.params.id
 
	playlist.get(username, trackid, (err, result) => {
		if (err) return res.send(400, err)
		return res.send(result)
	})
}
